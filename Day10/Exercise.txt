do
    for j in {1..10}
    do
        result=$((i * j))
        echo "$i * $j = $result"
    done
    echo "" 
done

Enter the no of rows:
5

echo "Enter the no of rows: "
read rows

for((i=1; i<=rows;i++))
do
for((j=i;j<rows;j++))
do
echo -n " "
done
for((k=1; k<=((2*i-1)); k++))
do
echo -n "*"
done
echo
done

for((i=rows-1; i>=1; i--))
do
for((j=rows-1;j>=i;j--))
do
echo -n " "
done
for((k=1;k<=((2*i-1));k++))
do
echo -n "*"
done
echo
done
